\begin{enumerate}[leftmargin=*]
%\setcounter{enumi}{1}
\item Se il Signore non costruisce la casa, * \mbox{invano vi faticano i costruttori.}
\item Se la città non è custodita dal Signore * \mbox{invano veglia il custode.}
\item Invano vi alzate di buon mattino, † tardi andate a riposare e mangiate pane di sudore: * \mbox{il Signore ne darà ai suoi amici nel sonno.}
\item Ecco, dono del Signore sono i figli, * \mbox{è sua grazia il frutto del grembo.} 
\item Come frecce in mano a un eroe * \mbox{sono i figli della giovinezza.}
\item Beato l’uomo * \mbox{che piena ne ha la farètra:}
\item non resterà confuso quando verrà alla porta * \mbox{a trattare con i propri nemici.}
\item Gloria al Padre e al Figlio * \mbox{e allo Spirito Santo.}
\item Come era nel principio, e ora e sempre * \mbox{nei secoli dei secoli. Amen.}
\end{enumerate}
\end{flushleft}
